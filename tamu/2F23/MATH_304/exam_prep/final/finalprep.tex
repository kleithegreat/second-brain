\section*{Question 1}
Find the general solution of each of the following systems:
\begin{enumerate}[a.]
    \item 
    $\begin{cases} 
    y_1 + y_2 = y'_1 \\
    -2y_1 + 4y_2 = y'_2 
    \end{cases}$
    
    \item 
    $\begin{cases} 
    y_1 - y_2 = y'_1 \\
    y_1 + y_2 = y'_2 
    \end{cases}$
    
    \item 
    $\begin{cases} 
    y_1 \quad\;\; + y_3 = y'_1 \\
    2y_2 + 6y_3 = y'_2 \\
    y_2 + 3y_3 = y'_3 
    \end{cases}$
\end{enumerate}
